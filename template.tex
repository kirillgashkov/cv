\documentclass[11pt]{article}


%
% I18n
%



% Multiple languages can be specified as a comma-separated list, the last one
% being the main. Example: "english,russian".
\usepackage[
  shorthands=off,
  $if(template.i18n.language)$
    $template.i18n.language$
  $else$
    english
  $endif$
]{babel}


%
% Layout
%


\usepackage{geometry}
\geometry{paper=a4paper}
\geometry{left=1.5cm,right=1.5cm,top=1.5cm,bottom=1.5cm}


%
% Font
%


\usepackage{fontspec}

% Computer Modern.
\setmainfont{CMU Serif}

% Provides \uline for underline and \sout for strikeout.
\usepackage[normalem]{ulem}


% Provides \enquote for language-aware quoting. Depends on babel.
\usepackage{csquotes}


% Provides \href for linking. hidelinks option removes borders around links. See
% https://tex.stackexchange.com/questions/823/remove-ugly-borders-around-clickable-cross-references-and-hyperlinks.
\usepackage[hidelinks]{hyperref}


%
% Paragraph
%


% Justifies the text.
\tolerance=1
\emergencystretch=\maxdimen
\hbadness=10000
\frenchspacing


%
% Hyphenation
%


% See https://tex.stackexchange.com/a/52009.
\makeatletter
\newcommand{\template@hyphenation@enable}{%
  \hyphenpenalty=50%
  \exhyphenpenalty=50%
}
\newcommand{\template@hyphenation@disable}{%
  \hyphenpenalty=10000%
  \exhyphenpenalty=10000%
}
\newcommand{\template@hyphenation@save}{%
  \mathchardef\template@hyphenation@internal@hyphenpenalty=\hyphenpenalty%
  \mathchardef\template@hyphenation@internal@exhyphenpenalty=\exhyphenpenalty%
}
\newcommand{\template@hyphenation@restore}{%
  \hyphenpenalty=\template@hyphenation@internal@hyphenpenalty%
  \exhyphenpenalty=\template@hyphenation@internal@exhyphenpenalty%
}
\makeatother

\makeatletter
\template@hyphenation@disable
\makeatother


%
% Header
%


% Removes section numbering.
\setcounter{secnumdepth}{0}


\usepackage{titlesec}
\titleformat{\section}[block]{\filright\scshape\large}{\thesection}{0.5em}{}[\titlerule]
\titlespacing*{\section}{0pt}{1.25em}{0.75em}


%
% List
%


\usepackage{enumitem}

% Any list.
\setlist{leftmargin=2\parindent,listparindent=\parindent,topsep=0.5em,partopsep=\parskip,parsep=\parskip,itemsep=\parskip}

% Unordered list.
\setlist[itemize]{label=\raisebox{0.2ex}{\scriptsize$$\bullet$$}}

% Ordered list.
\setlist[enumerate,1]{label={\arabic*.}}
\setlist[enumerate,2]{label={\arabic*.}}
\setlist[enumerate,3]{label={\arabic*.}}
\setlist[enumerate,4]{label={\arabic*.}}

% Defines \tightlist for Pandoc.
\newcommand{\tightlist}{}


%
% Math
%


% Provides macros for typesetting math. The packages were chosen to emulate
% KaTeX. See https://github.com/KaTeX/KaTeX/wiki/Package-Emulation.
\usepackage{amsmath}
\usepackage{amsfonts}
\usepackage{amssymb}


% Provides support for Unicode math fonts.
\usepackage{unicode-math}

\setmathfont{Latin Modern Math}


%
% Other
%


% tabularx allows for more flexible table column widths.
\usepackage{tabularx}

% \st - strikeout (generated by Pandoc).
\usepackage{soul}

% Disables page numbering.
\pagenumbering{gobble}

$body$

\end{document}
